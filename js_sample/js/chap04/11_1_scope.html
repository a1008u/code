スコープ①

scope = 'Global Variable';
function getValue() {
 console.log(scope);
 var scope = 'Local Variable';
 return scope;
}

console.log(getValue());


var scope3 = 'Global Variable';
function getValue() {
 console.log(scope3);
 var scope3 = 'Local Variable';
 return scope3;
}

console.log(getValue());
console.log(scope3);


function getValue2() {
 console.log(scope2); // 変数の巻き上げ
 var scope2 = 'Local Variable';
 return scope2;
}

console.log(getValue2()); console.log(scope2);

特徴

  1. ブラウザ上のグローバルオブジェクト = Windowオブジェクト
  2. 関数スコープいない変数はvarを付けようがつけまいが、必ずグローバルオブジェクト(Windowオブジェクト)に保持される
