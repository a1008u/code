引数

let member = {
 mid: 'Y0001',
 name: '山田太郎',
 address: 't_yamada@example.com'
};

function show({name}) {
 console.table(name);
};
show(member);

function show2({mid, name}) {
 console.log(`mid === ${mid} name === ${name}`);
};
show2(member);

function show3(name) {
 console.table(name);
};
show3(member);

特徴

  1. 引数にJsonを利用していると、Jsonのkeyを仮引数に設定するとvalueを取得できる
